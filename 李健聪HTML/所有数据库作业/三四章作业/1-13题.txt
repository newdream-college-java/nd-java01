3查询SQL考试的最低分数
SELECT MIN(examresult)'SQL最低分' from studentexam WHERE examsubject='SQL' ; 
4查询Java考试成绩最高的学员姓名
SELECT max(examresult)'JAVA最高分' from studentexam WHERE examsubject='java' ; 
5查询学院‘火云邪神’一共参加了几次考试
SELECT COUNT(*) from studentexam s,studentinfo1 t WHERE s.estuid=t.stuid and t.stuname='火云邪神';
6.查询各科目的平均成绩 
SELECT avg(examresult) from studentexam GROUP BY examsubject;
7.查询每个班级学员的最小年龄

8.查询考试不及格的人数
SELECT COUNT(*) from studentexam s where  examresult<60 GROUP BY estuid; 
9.查询各学员的总成绩，要求筛选出总成绩在140分以上的
SELECT sum(stu.examresult),st.stuname from studentexam stu,studentinfo1 st WHERE stu.estuid=st.stuid GROUP BY stu.estuid HAVING sum(stu.examresult)>140;
10.查询男女学员的平均年龄
SELECT avg(stu.examresult),st.stusex from studentexam stu,studentinfo1 st  WHERE stu.estuid=st.stuid GROUP BY st.stusex;

select ExamSubject,avg(ExamResult)'平均分' from studentexam GROUP BY  ExamSubject HAVING  avg(ExamResult)>=80 ;
